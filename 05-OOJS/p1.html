Literal Object

key: value => property of an object

SYNTAX :

        let literalObjectName = {
            "key1":"value1",
            "key2":"value2",
            "key3":"value3",
            "key4":"valueN",
        };

        console.log(literalObjectName);
        console.log(typeof(literalObjectName));
        console.log(literalObjectName.key1);
